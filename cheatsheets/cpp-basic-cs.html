Basic Usage of Reference and Pointer

#include <iostream>
using namespace std;

int main(int argc, char* argv[])
{
   int a = 5566;
   int *p_a_addr = &a;
   int &r_a_addr = a;

   /* value of a*/
   cout << a << endl;
   /* address of a */
   cout << &a <<endl;
   /* address of a */
   cout << p_a_addr << endl;
   /* address of pointer */
   cout << &p_a_addr <<endl;
   /* value of a */
   cout << r_a_addr << endl;
   /* address of a */
   cout << &r_a_addr << endl;
   return 0;
}

/* 
output:

5566
0x7fff586ef6ec
0x7fff586ef6ec
0x7fff586ef6e0
5566
0x7fff586ef6ec
*/

Lambda Function

/* Lambda since C++11 */
#include <iostream>
using namespace std;

int main(int argc, char *argv[])
{
   int a = 8;
   int sa = 0;

   /* lambda square(since c++11) */
   sa = [](int x) {return x*x;}(a);
   cout << "a = " << a << endl;
   cout << "square a = "
        << sa << endl;
   
   return 0;
}
/*
bash> g++ ex.cpp --std=c++11
output:

a = 8
square a = 64
*/

Template in C++

#include <iostream>
using namespace std;

template <class T> struct Cls {
   T var;
   Cls(T in):var(in) {
      var = in; 
   };
   void spam() {
      cout << var << endl;
   };
};

int main(int argc, char *argv[])
{
   Cls<int> obj1(5566);
   Cls<double> obj2(95.27);
   
   obj1.spam();
   obj2.spam();

   return 0;
}
/* 
output:

5566
95.27
*/

Class "operator" Overloading

#include <iostream>
using namespace std;

class cls {
public:
   /* obj++ operation */
   cls operator++(int) {
      cls tmp(idx);
      idx++;
      return tmp; 
   };
   /* ++obj operation */
   cls& operator++() {
      ++idx;
      return *this;
   };
   void spam() {
      cout << "idx val = "
           << idx << endl;
   };
   cls(int x) :idx(x) {};
private:
   int idx;
};

int main(int argc, char *argv[])
{
   cls obj(5566);

   obj.spam();
   ++obj;
   obj.spam();
   obj++;
   obj.spam();
   return 0;
}
/*
output:

idx val = 5566
idx val = 5567
idx val = 5568
*/

Different between Pointer and Reference

#include <iostream>
using namespace std;

int main(int argc, char *argv[])
{
   int a = 5566;
   int b = 9527;
   int *ptr1, *ptr2;
   int &ref1 = a; 
   int &ref2 = b;

   ptr1 = &a;
   ptr2 = &b;
   cout << "b value will not change" 
        << endl;
   cout << "val b = "
        << b  << endl;
   ptr2 = &a;
   cout  << "val b = "
         << b  << endl;

   cout  << "b value will change"
         << endl;
   cout  << "val b = "
         << b  << endl;
   ref2 = a;
   cout  << "val b = "
         << b  << endl;
   
   return 0;
}

/* 
output:

b value will not change
val b = 9527
val b = 9527
b value will change
val b = 9527
val b = 5566
*/

C++ struct

/* C++ struct is same as class
 * which all members are public.
 */
#include <iostream>
using namespace std;

struct cppstruct {
   int a;
   double b;
   void spam(){
     cout << a << endl; 
     cout << b << endl;
   };
};

class cls {
public:
   int a;
   int b;
   void spam(){
      cout << a << endl;
      cout << b << endl;
   };
};

int main(int argc, char *argv[])
{
   cppstruct obj1;
   cls obj2;

   cout << "cpp strcut" 
        << endl;
   obj1.a = 5566;
   obj1.b = 9527;
   obj1.spam();

   cout << "all member public class"
        << endl;
   obj2.a = 1234;
   obj2.b = 5677;
   obj2.spam();
   
   return 0;
}

/*
output:

cpp strcut
5566
9527
all member public class
1234
5677
*/

Call by "Reference"

#include <iostream>
using namespace std;

void spam_ptr(int *ptr)
{
   cout << "addr = " 
        << ptr << endl; 
   cout << "val = "
        << *ptr << endl;
}

void spam_ref(int &ref)
{
   cout << "addr = "
        << &ref << endl;
   cout << "val = "
        << ref <<endl;
}

int main(int argc, char *argv[])
{
   int var = 5566;
   spam_ptr(&var);
   spam_ref(var);
   
   return 0;
}
/*
output:

addr = 0x7fff53c6c74c
val = 5566
addr = 0x7fff53c6c74c
val = 5566
*/

Object Constructor

#include 
using namespace std;

class cls1 {
public:
   int x;
   int y;
   /* Basic Constructor */
   cls1(int a, int b) {
      cout << "Construct cls1" 
           << endl;
      x=a;
      y=b;
   };
   ~cls1() {
      cout << "Destruct cls1" 
           << endl;
   };
};

class cls2 {
public:
   int x;
   int y;
   /* Initialization Lists */
   cls2(int a, int b):x(a), y(b) {
      cout << "Construct cls2"
           << endl;
   }
   ~cls2() {
      cout << "Destruct cls2" 
           << endl;
   };
};

int main(int argc, char *argv[])
{
   cls1 obj1(5566, 9527);
   cls2 obj2(123,456);
   cout << obj1.x << ", "
        << obj1.y << endl;
   cout << obj2.x << ", "
        << obj2.y << endl;
   return 0;
}
/* 
output:

Construct cls1
Construct cls2
5566, 9527
123, 456
Destruct cls2
Destruct cls1
*/
