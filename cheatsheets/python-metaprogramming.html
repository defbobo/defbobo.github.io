Debuging via Decorator

# Using print is ugly.
>>> def spam1():
...   print spam1.__name__
...   print 'SPAM_1'
... 
>>> def spam2():
...   print spam2.__name__
...   print 'SPAM_2'
... 
>>> spam1()
spam1
SPAM_1
>>> spam2()
spam2
SPAM_2

# Using decorator
>>> def debug(func):
...   def wrapper(*args, **kwargs):
...     print func.__name__
...     return func(*args, **kwargs)
...   return wrapper
... 
>>> @debug
... def spam():
...   print 'SPAM'
... 
>>> spam()
spam
SPAM

@wraps - Get Original Function's "Description"

>>> from functools import wraps
>>> def debug(func):
...     @wraps(func)
...     def wrapper(*args, **kwargs):
...         print func.__doc__
...         print func.__name__
...         return func(*args, **kwargs)
...     return wrapper
... 
>>> @debug
... def spam():
...     '''spam function'''
...     print "SPAM"
... 
>>> spam()
spam function 
spam
SPAM

Decorate "class" methods

>>> from functools import wraps
>>> def debug(func):
...     @wraps(func)
...     def wrapper(*args, **kwargs):
...         print "doc:", func.__doc__
...         print "func: ", func.__name__
...         return func(*args, **kwargs)
...     return wrapper
... 
>>> def debugmethods(cls):
...     for k,v in vars(cls).items():
...         if callable(v):
...             setattr(cls, k, debug(v))
...     return cls
...
>>> @debugmethods
... class Spam(object):
...     def hello(self):
...         ''' I am hello '''
...         print "Hello"
...     def world(self):
...         ''' I am world '''
...         print "World"
... 
>>> s = Spam()
>>> s.hello()
doc:  I am hello 
func:  hello
Hello
>>> s.world()
doc:  I am world 
func:  world
World
