Cheat sheet of Python. Some basic concept for debug, test and profiling.

Debug Switch - logging

# see: python document-Logging HOWTO
# "print" debug message when you need.
# default level: warning
>>> import logging
>>> logging.debug("debug")
>>> logging.info("info")
>>> logging.warning("warning")
WARNING:root:warning
>>> logging.error("error")
ERROR:root:error
>>> logging.critical("cirtical")
CRITICAL:root:cirtical

# set up level: debug
>>> logging.basicConfig(
...   level =logging.DEBUG)
>>> logging.debug("debug")
DEBUG:root:debug
>>> logging.info("info")
INFO:root:info
>>> logging.warning("warning")
WARNING:root:warning
>>> logging.error("error")
ERROR:root:error
>>> logging.critical("cirtical")
CRITICAL:root:cirtical

# log to a file(raw data)
>>> logging.basicConfig(
...   filename='example.log',
...   level=logging.DEBUG)

# display with format with date
>>> logging.basicConfig(
...   format='%(asctime)s: %(message)s',
...   datefmt='%m/%d/%Y %I:%M:%S %p')
>>> logging.warning("warning")
03/16/2015 10:46:35 PM: warning

# logger object
>>> logger = logging.getLogger(__name__)
>>> logger.setLevel(logging.DEBUG)
>>> logger.info('info')
INFO:__main__:info
# set logger level
>>> logger.setLevel(logging.ERROR)
>>> logger.info('info')

Test Cases in "unittest"

# basic
assertEqual(a,b)          # a==b
assertNotEqual(a,b)       # a!=b
assertTrue(x)             # bool(x) is True
assertFalse(x)            # bool(x) is False
assertIs(a,b)             # a is b
assertIsNot(a,b)          # a is not b
assertIsNone(x)           # x is None
assertIsNotNone(x)        # x is not None
assertIn(a,b)             # a in b
assertNotIn(a,b)          # a not in b
assertIsInstance(a,b)     # isinstance(a,b)
assertNotIsInstance(a,b)  # not isinstance(a,b)

# raise
assertRaises(exc, fun, *args, **kwds)
assertRaisesRegexp(exc, r, fun, *args, **kwds)

# perform specific check
assertAlmostEqual(a,b)        # round(a-b, 7) == 0
assertNotAlmostEqual(a,b)     # round(a-b, 7) != 0
assertGreater(a,b)            # a > b
assertGreaterEqual(a,b)       # a >= b
assertLess(a,b)               # a < b
assertLessEqual(a,b)          # a <= b
assertRegexpMatches(s,r)      # r.search(s) 
assertNotRegexpMatches(s,r)   # not r.search(s)
assertItemsEqual(a,b)         # sorted(a) == sorted(b)
assertDictContainsSubset(a,b) # all the key/value pairs in a exist in b

# others
assertMultiLineEqual(a, b)
assertSequenceEqual(a, b)
assertListEqual(a, b)
assertTupleEqual(a, b)
assertSetEqual(a, b)
assertDictEqual(a, b)

Calculate Time Cost of a Function

>>> def cal_time(func):
...   def wrapper(*args, **kwargs):
...     start = time.time()
...     func(*args, **kwargs)
...     end   = time.time()
...     print "cost time:", end - start 
...   return wrapper
... 
>>> @cal_time
... def sleep_5_sec():
...   time.sleep(5)
... 
>>> sleep_5_sec()
cost time: 5.00077295303
>>> @cal_time
... def runloop():
...   for _ in range(100000):
...     pass
... 
>>> runloop()
cost time: 0.00877285003662

Profiling - How Efficiency of a Function

>>> import cProfile
>>> import pstats
>>> def profiling(func):
...   def wrapper(*args, **kwargs):
...     profile = cProfile.Profile()
...     profile.enable()
...     func(*args, **kwargs)
...     profile.disable()
...     pstat = pstats.Stats(profile)
...     pstat.print_stats()
...   return wrapper
... 
>>> @profiling 
... def loop_sleep():
...   for _ in range(100):
...     time.sleep(0.01)
...

Profiling Per Line Via Line Profiler API

# need install line_profiler first
# see: https://github.com/rkern/line_profiler
# pip install line_profiler
# example.py
from line_profiler import LineProfiler
import time

def profiling(func):
   def wrapper(*args, **kwargs):
      profiler = LineProfiler(func)
      profiler.enable_by_count()
      ret = func(*args, **kwargs)
      profiler.print_stats()
      return ret
   return wrapper

@profiling
def sleep_loop():
   for _ in range(100):
      time.sleep(0.01)
sleep_loop()
